#include <stdio.h>
#include <stdlib.h>

#define elementy 15*15

void zapelnij(int *array)
{
int i;
    for(i=0; i<elementy; i++)
    {
        array[i]=0;
    }
}

void wypisz(int *array)
{
int i;
    for(i=0; i<elementy; i++)
    {
        printf("%d ",array[i]);
    }
    puts("");
}

int main(void)
{
int *array_pointer = (int *)calloc(elementy, sizeof(int));
if(array_pointer)
{
    zapelnij(array_pointer);
    wypisz(array_pointer);
    free(array_pointer);
    array_pointer = NULL;
}
return 0;
}
